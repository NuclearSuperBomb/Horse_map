const imgName = 'J10RailroadEngineerRDR2MapLight24.jpg'; function initMap() { if (typeof L === 'undefined') return; // Configurazione del Sistema di Coordinate Personalizzato (RDR2 Scale) // Trasformiamo i pixel in modo che corrispondano ai tuoi valori negativi/positivi const factor = 0.1; // Fattore di scala per gestire i numeri grandi di RDR2 const map = L.map('map', { crs: L.Util.extend({}, L.CRS.Simple, { transformation: new L.Transformation(factor, 0, -factor, 0) }), minZoom: -5, maxZoom: 2 }); // Punti di riferimento per calcolare i confini (basati sui tuoi dati) // Abbiamo bisogno di definire un'area che contenga tutti i tuoi punti reali const southWest = L.latLng(-6000, -3000); const northEast = L.latLng(4000, 3000); const bounds = new L.LatLngBounds(southWest, northEast); // Carichiamo l'immagine adattandola ai nuovi confini reali L.imageOverlay(imgName, bounds).addTo(map); map.fitBounds(bounds); // 2. I TUOI PUNTI CON LE COORDINATE ORIGINALI RDR2 const points = [ { name: "Punto 1", coords: [-2020.76, -5180.16] }, // Nota: Invertiti in [Y, X] { name: "Punto 2", coords: [-2233.96, 2475.32] }, { name: "Punto 3", coords: [2380.70, 2906.16] } ]; const listDiv = document.getElementById('point-list'); points.forEach(p => { const marker = L.marker(p.coords).addTo(map).bindPopup(p.name); const item = document.createElement('div'); item.className = 'list-item'; item.innerText = p.name; item.onclick = () => { map.setView(p.coords, 0); marker.openPopup(); }; listDiv.appendChild(item); }); // 3. LOG COORDINATE (Ora vedrai i valori nello stile RDR2!) map.on('click', function(e) { // Mostra X (Lng) e Y (Lat) come richiesto console.log("X: " + e.latlng.lng.toFixed(4) + ", Y: " + e.latlng.lat.toFixed(4)); }); } window.onload = initMap;
